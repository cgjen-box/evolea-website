export const ORG_ID = 'https://www.evolea.ch/#organization';

export function siteGraph(siteSettings: unknown, lang: 'de' | 'en', site: URL | undefined, base: string): Record<string, unknown>;

// items match InnerPageHero's BreadcrumbItem shape: { label: string; href?: string } — last item (current page) omits href

export function blogPostingSchema(opts: {
  headline: string; description: string; datePublished?: string; // ISO string, omit when invalid
  authorName: string; imageUrl?: string; canonicalUrl: string; lang: 'de' | 'en';
}): Record<string, unknown>;

export function serviceSchema(opts: {
  name: string; description: string; url: string; lang: 'de' | 'en';
  minAge?: number; maxAge?: number; // omit audience block entirely when undefined
}): Record<string, unknown>;

export function cafeEventSchema(opts: { lang: 'de' | 'en'; description: string }): Record<string, unknown>;

// Next "2nd Wednesday of a month, 20:00" strictly in the future, e.g. '2026-07-08T20:00:00+02:00'
export function nextCafeDate(now?: Date): string;
